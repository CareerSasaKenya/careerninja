```bash
cd careerninja

# Check status
git status

# Add all Phase 5.1 files
git add supabase/migrations/20260307_candidate_career_tools_fixed.sql
git add src/lib/careerTools.ts
git add app/dashboard/career-tools/
git add src/components/career-tools/
git add src/components/dashboards/CandidateDashboard.tsx
git add PHASE_5.1_CAREER_TOOLS.md

# Commit with descriptive message
git commit -m "feat: Phase 5.1 - Candidate Career Tools

- CV Builder with multiple versions and templates
- Cover Letter Generator with smart placeholders
- Skill Assessments with scoring and certificates
- Career Path Planner with AI suggestions
- Salary Insights with market comparison
- Complete database schema with RLS policies
- Comprehensive documentation

Includes:
- 10 new database tables
- 5 major UI components
- TypeScript library with full CRUD operations
- Seed data for templates and sample data
- Integration with candidate dashboard"

# Push to remote
git push origin main
# Or if you're on a different branch:
# git push origin your-branch-name
```
